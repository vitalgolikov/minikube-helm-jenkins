# Prepare my minikube to test

   Create namestape
   kubectl apply -f jenkins-namespace.yaml
   Create Persistant volume
   kubectl apply -f jenkins-volume.yaml
   Create Service Account
   kubectl apply -f jenkins-sa.yaml

# Deploy with helm

    helm show values jenkins/jenkins
    helm show values jenkins/jenkins > values.yaml
    change values persistence:
    change values serviceAccount:
    set to my minikube test ServiceType  NodePort

    helm install jenkins -n jenkins -f values.yaml jenkins/jenkins
